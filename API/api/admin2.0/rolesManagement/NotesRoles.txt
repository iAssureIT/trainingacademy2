=====================================================================================
1. Insert Role Details
=====================================================================================

	Method  : POST

	URL 	: '/api/roles/post'

	Input	: 
				{
				    "role"        : String,
				    "createdBy"   : String,
				}

	Output	: If Successfully
				{
					message : "ROLE_ADDED",
					ID 		: String //_id of the document
				}
			  If not successful
			  	{
			  		message : "ALL_ARE_REQUIRED_FIELDS:keys,secret,bucket,region",
			  	}

=====================================================================================
2. Fetch Role Details
=====================================================================================

	Method  : GET

	URL 	: '/api/roles/get/:ID'

	Input	: - NA -
				

	Output	: If Successfully
				{
				    _id			: String,
				    role        : String,
				    createdAt   : Date,
				    createdBy   : String
				}
			  If not successful
			  	{
			  		message : "ROLE_NOT_FOUND",
			  	}

=====================================================================================
